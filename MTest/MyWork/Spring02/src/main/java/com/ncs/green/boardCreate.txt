1. Board Table 생성 & 기본자료 입력

drop table board;
=> Table Create

create table board (    
seq int(5) primary key AUTO_Increment,
id varchar(10) not null,
title varchar(200) not null,
content Text(2000),
regdate  datetime DEFAULT CURRENT_TIMESTAMP,
cnt int default 0
);

=> Date 5개 이상 입력
